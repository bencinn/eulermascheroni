\begin{lemma}[Nonnegativeness of the inner sum]
  \label{lem:eulerMascheroni_inner_nonneg}
  \lean{eulerMascheroni_sum_inner_nonneg}
  \leanok
  $\frac{1}{n} - \ln{\frac{n+1}{n}} \geq 0$ for all $n$ greater than zero.
\end{lemma}

\begin{proof}
  Since $\ln x \leq x - 1$ for all $x$ greater than $0$ (see "log_le_sub_one_of_pos" in mathlib4), and $\frac{n+1}{n} \gt 0$ for all $n$ greater than $0$, we can rewrite as $\frac{1}{n} \geq \ln{\frac{n+1}{n}} \geq \frac{n+1}{n} - 1 = \frac{1}{n}$.
\end{proof}

\begin{lemma}[Partial sum of the infinite sum]
  \label{lem:eulerMascheroni_tsum_partial}
  \lean{eulerMascheroni_sum_partial}
  \leanok
  $\sum^{N}_{n=1}{\left(\frac{1}{n} - \ln{\frac{n+1}{n}}\right)} = H_n - \ln{\left( N+1 \right)}$ for all $N$ greater than 0
\end{lemma}

\begin{proof}
  We can separate the sum into $\sum^{N}_{n=1}{\frac{1}{n}}$ (which is $H_n$) and $-\sum^{N}_{n=1}{\left(\ln{\frac{n+1}{n}}\right)} = -\sum^{N}_{n=1}{\ln{n+1} - \ln{n}}$, which is a telescopic sum that results in $-(\ln{N+1} - \ln{1}) = -\ln{N+1}$.
\end{proof}
